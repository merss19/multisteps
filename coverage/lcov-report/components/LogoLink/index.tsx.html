import * as React from 'react'
import {browserHistory} from 'react-router'
import cookie from 'react-cookie'
 
interface OwnProps {
	isTest?: boolean,
}

const LogoLink: React.SFC<OwnProps> = ({isTest}) => (
	<svg className='svg-icon ys-logo-web'
		 onClick={() => {
			 cookie.remove('token', {path: '/'})
			 cookie.remove('txId', {path: '/'})
			 cookie.remove('role', {path: '/'})
			 cookie.remove('program', {path: '/'})
			 cookie.remove('packageType', {path: '/'})
			 cookie.remove('promoName', {path: '/'})
			 cookie.remove('share', {path: '/'})
			 cookie.remove('userProgram', {path: '/'})
			 cookie.remove('tester', {path: '/'})
			 cookie.remove('abtest', {path: '/'})
			 if (isTest) {
				 window.location.href = 'https://gohero.todayme.ru/'
			 } else {
				 browserHistory.push('/')
			 }
 
		 }}>
		<use xlinkHref='#ys-logo-web'/>
	</svg>)
 
export default LogoLink
