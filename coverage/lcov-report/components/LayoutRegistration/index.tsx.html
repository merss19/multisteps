import * as React from 'react'
import {connect} from 'react-redux'
import Background from '../Background'
import LogoLink from '../LogoLink'
import * as SVGInjector from 'svg-injector'
console.log(process)
console.log('processs')
interface Sprite {
	readonly body: any
	readonly img: HTMLImageElement
}
let url = process.env.NODE_ENV === 'test' ? 'https://lk.todayme.ru/' : '/'
class LayoutRegistration extends React.Component<{}, {}> {
 
	componentDidMount() {
		this.svgInject()
	}
 
	svgInject(): void {
		const sprite: Sprite = {
			body: document.querySelector('body'),
			img : document.createElement('img')
		}
		sprite.img.className = 'injected-svg'
		sprite.img.id = 'svg-inject-me'
		sprite.img.src = url + 'assets/img/symbol-sprite.svg'
		sprite.body.appendChild(sprite.img)
		let mySVGsToInject: any = document.querySelector('#svg-inject-me')
		SVGInjector(mySVGsToInject)
	}
 
	render() {
		const {children} = this.props
		return (
			<div className='layout layout--entry'>
				<div className='grid entry-header'>
					<div className='grid__cell todayme-logo'>
						<LogoLink/>
					</div>
				</div>
				{children}
				<Background />
			</div>
		)
	}
}
 
export default LayoutRegistration
