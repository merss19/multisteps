export const selectPrograms = createSelector(
	programs,
	choosenProgram,
	(programs, choosenProgram) => {
		programs = programs.map(program => {
			if (choosenProgram == 0 ) {
				if (program.get('id') % 4 == 1) {
					return program.set('isActive', true)
				}
			}
			if (program.get('id') == choosenProgram) {
				return program.set('isActive', true)
			}
			return program
		})
		return programs.toJS()
	}
)
