export const paymentCreate = (data: Payment, cb) => dispatch => {
	dispatch(request())
	fetchPaymentCreate(data)
		.then(response => response.json())
		.then(res => {
			if (res.data && res.errorCode === 1) {
				dispatch(success())
				dispatch(getPayment(res.data))
				cb()
			} else {
				dispatch(failure())
			}
		})
		.catch((err) => {
			throw new Error(err.message)
		})
}
export const getPackage = (data?: string) => dispatch => {
	dispatch(request())
		fetchGetPromo(data)
			.then(response => response.json())
			.then(res => {
				if (res.data && res.errorCode === 1) {
					let packages = res.data
					packages = packages.map(item => {
						return {
							...item,
							isActive: false
						}
					})
					dispatch(success())
					dispatch(getPackages(packages))
				} else if (res.errorCode === 3) {
					dispatch(success())
					dispatch(promoError(res.errorMessage))
				} else {
					dispatch(failure())
				}
			})
			.catch((err) => {
				throw new Error(err.message)
			})
}

export default (state = initialStateImmutable, action) => {
	switch (action.type) {
		case CHOOSEN_PROGRAM:
			return state.set('choosenProgram', action.data)
		case SET_PACKAGE_TYPE:
			return state.set('choosenPackageType', action.data)
		case GET_PAYMENT:
			return state.merge({
				payment: action.data
			})
		case GET_PACKAGES:
			return state.merge({
				packages: action.data,
				promoError: '',
			})
		case SET_PROMO:
			return state.merge({
				promo: action.data
			})
		case PROMO_ERROR:
			return state.merge({
				promoError: action.data
			})
 
		default:
			return state
	}
}
