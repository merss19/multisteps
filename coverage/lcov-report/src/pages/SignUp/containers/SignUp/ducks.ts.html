import * as Immutable from 'immutable'
import { combineReducers } from 'redux'
import fetchData from './api'
import {Action, StateSignUp} from './interfaces'
import {Steps} from './const'
import { step1Reducer } from '../Step1'
import { stepTwoReducer } from '../Step2'
import { stepThreeReducer } from '../Step3'
// Constant
const RECEIVE_PROGRAMS = 'RECEIVE_PROGRAMS'
const RECEIVE_PROGRAMS_FAIL = 'RECEIVE_PROGRAMS_FAIL'
const CHANGE_STEP = 'CHANGE_STEP'
// Action Creators

const success = (data): Action => ({
	type: RECEIVE_PROGRAMS,
	data
})

const failure = (): Action => ({
	type: RECEIVE_PROGRAMS_FAIL,
})

export const fetchPrograms = () => dispatch => {
	// dispatch(request())
	console.log('fetchProgramssss')
	fetchData()
		.then(response => response.json())
		.then(res => {
			console.log(res)
			console.log('ressss')
			if (res.errorCode === 1) {
				let programs = res.data
				programs = programs.map(program => {
					return {
						...program,
						isActive: false
					}
				})
				dispatch(success(programs))
			} else {
				dispatch(failure())
			}
		})
		.catch((err) => {
			throw new Error(err.message)
		})
}

export const changeStep = (data: number) => dispatch => {
	console.log('changeSteppp')
	dispatch({
		type: CHANGE_STEP,
		data
	})
}
 
// Reducer
const initialState: StateSignUp = {
	step: Steps.two,
	isTest: false,
	isTele2: false,
	isInfoBlock: false,
	programs: []
}
 
const initialStateImmutable = Immutable.fromJS(initialState)

const signUpReducer = (state = initialStateImmutable, action) => {
	switch (action.type) {
		case RECEIVE_PROGRAMS:
			return state.merge({
				programs: action.data
			})
		case CHANGE_STEP:
			return state.set('step', action.data)

		default:
			return state
	}
}

const  registrationReducer = combineReducers({
	step1: step1Reducer,
	signUp: signUpReducer,
	stepTwo: stepTwoReducer,
	stepThree: stepThreeReducer
})
 
export default registrationReducer
