import * as Immutable from 'immutable'
import { fetchUserCreate } from './api'
import {Action, SignUpFormData, Step1} from './interfaces'
// Constant
const SIGNUP_REQUEST = 'SIGNUP_REQUEST'
const SIGNUP_SUCCESS = 'SIGNUP_SUCCESS'
const SIGNUP_FAILURE = 'SIGNUP_FAILURE'
const CHANGE_STEP = 'CHANGE_STEP'
const SET_RADIO_ERROR = 'SET_RADIO_ERROR'
const REMOVE_RADIO_ERROR = 'REMOVE_RADIO_ERROR'
// Action Creators
const request = (): Action => ({
	type: SIGNUP_REQUEST,
})
const success = (data): Action => ({
	type: SIGNUP_SUCCESS,
	data
})
 
const failure = (): Action => ({
	type: SIGNUP_FAILURE,
})

export const userCreate = (data: SignUpFormData, cb) => dispatch => {
	dispatch(request())
	fetchUserCreate(data)
		.then(response => response.json())
		.then(res => {
			console.log(res)
			console.log('userCreateeee')
			const data = res.data
			if (data && res.errorCode === 1) {
				dispatch(success(data))
				cb(data)
			} else {
				dispatch(failure())
			}
		})
		.catch((err) => {
			throw new Error(err.message)
		})
}

export const changeStep = (data: number) => dispatch => {
	console.log('changeSteppp')
	dispatch({
		type: CHANGE_STEP,
		data
	})
}

export const setRadioError = (data) => dispatch => {
	console.log('SET_RADIO_ERROR')
	dispatch({
		type: SET_RADIO_ERROR,
		data
	})
}
export const removeRadioError = (data) => dispatch => {
	console.log('RADIO_ERROR_REMOVE')
	dispatch({
		type: REMOVE_RADIO_ERROR,
		data
	})
}
// Reducer
const initialState: Step1 = {
	isErrorGender: false,
	genderError: ''
}

const initialStateImmutable = Immutable.fromJS(initialState)
 
export default (state = initialStateImmutable, action) => {
	switch (action.type) {
		case SET_RADIO_ERROR:
			return state.set('isErrorGender', true)
									.set('genderError', action.data)
		case REMOVE_RADIO_ERROR:
			return state.set('isErrorGender', false)
									.set('genderError', '')
		default:
			return state
	}
}
