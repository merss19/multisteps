import * as React from 'react'
import {connect} from 'react-redux'
import {Field, FormProps, reduxForm} from 'redux-form'
import {Link} from 'react-router'
import {selectIsError} from './selectors'
import InputText from '../../../../components/InputText'
import Button from '../../../../components/Button'

interface LoginFormData {
	email: string
	password: string
}

interface OwnProps extends FormProps<LoginFormData, {}, {}> {
	onSubmit: (values: LoginFormData) => void
}
type ConnectedState = {
	isError: boolean;
}
type ConnectedDispatch = {}
//selectors(state)
const mapStateToProps = (state, ownProps: OwnProps): ConnectedState => ({
	isError: selectIsError(state)
})

const mapDispatchToProps = (dispatch): ConnectedDispatch => ({})

class LoginFormComponent extends React.Component<ConnectedState & OwnProps, {}> {
	render() {
		const {onSubmit, isError, handleSubmit} = this.props
		return (
			<div className='entry entry--min'>
				<div className='entry__inner'>
					{ isError ? <div>Неправильный email или пароль</div> : null }
 
					<div className='entry__header'>
						<h2 className='entry__title text-center'>Вход в Личный кабинет</h2>
					</div>
 
					<form onSubmit={handleSubmit!(onSubmit)} className='entry__box'>
						<Field name='email'
									 placeholder='Ваш e-mail'
									 type='email'
									 component={InputText}
						/>
						<Field name='password'
									 placeholder='Ваш пароль'
									 type='password'
									 component={InputText}
						/>

						<Button
							type={'submit'}
							btnType={'secondary'}
							wide={true}
						>
							Войти
						</Button>
 
						<ul className='entry-nav mtb20'>
							<li className='entry-nav__item'>
								<Link to='/signup'>Регистрация</Link>
							</li>
							<li className='entry-nav__item'>
								<Link to='/restore'>Забыли пароль?</Link>
							</li>
						</ul>
 
					</form>
 
				</div>
			</div>
		)
	}
}
 
const validate = (data) => {
	const errors: any = {}
 
	if (!data.email) {
		errors.email = 'Email должен быть заполнен'
	}
	if (!/^([\w-]+(?:\.[\w-]+)*)@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$/i.test(data.email)) {
		errors.email = 'Email заполнен неправильно'
	}
	if (!data.password) {
		errors.password = 'Поле пароля должно быть заполнено'
	}
	return errors
}
 
const LoginForm = connect(mapStateToProps)(LoginFormComponent)
 
export default reduxForm({
	form: 'LoginForm',
	validate
})(LoginForm)
