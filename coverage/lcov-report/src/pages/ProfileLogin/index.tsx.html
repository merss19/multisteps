import * as React from 'react'
import {connect} from 'react-redux'
import LoginForm from './containers/LoginForm'
import  reducer, {submitLogin} from './ducks'
export const profileLoginReducer = reducer
import {LoginFormData} from './interfaces'
import {browserHistory} from 'react-router'
import * as SVGInjector from 'svg-injector'
import Background from '../../components/Background'
import LogoLink from '../../components/LogoLink'
import Cookie from '../../tools/cookie'
import Carrot from '../../tools/carrot'
import  './style.scss'
console.log(SVGInjector)

type ConnectedDispatch = {
	submitLogin(data: LoginFormData, cb: (data: any) => void)
}

class ProfileLoginComponent extends React.Component<ConnectedDispatch, {}> {

	componentDidMount() {
		var mySVGsToInject: any = document.querySelector('#svg-inject-me')
		console.log('didd')
		console.log(mySVGsToInject)

		// Do the injection
		SVGInjector(mySVGsToInject)
	}

	onSubmit = (data: LoginFormData) => {
		this.props.submitLogin(data, this.resultAction)
	}
 
	resultAction(data: any): void {
		if (data.paidState !== 0) {
			Carrot().auth(data.id, data.cqhmac)
			Cookie().save('token', data.authToken, {path: '/', maxAge: 60 * 60 * 24 * 365 * 10})
			// setToken(data.authToken)
			Cookie().save(
				'userProgram', data.program, {path: '/', maxAge: 60 * 60 * 24 * 365 * 10}
			)
			Cookie().save(
				'fullName', data.firstName + ' ' + data.lastName,
				{path: '/', maxAge: 60 * 60 * 24 * 365 * 10}
			)

			if (data.paidState === 2 && (data.program !== 12 && data.program !== 16)) {
				browserHistory.push('/signup/pay/success')
			} else if (data.paidState === 2 && (data.program === 12 || data.program === 16)) {
				browserHistory.push('/signup/pay/friend?simple=true')
			} else if (data.isFirstEdit) {
				browserHistory.push('/task')
			} else {
				browserHistory.push('/profile')
			}
		} else if (data.paidState !== 0 && (data.program === 12 || data.program === 16)) {
			Carrot().auth(data.id, data.cqhmac)
			Cookie().save('token', data.authToken, {path: '/', maxAge: 60 * 60 * 24 * 365 * 10})
			browserHistory.push('/signup/pay/friend?simple=true')
		} else if (!data.program) {
			Carrot().auth(data.id, data.cqhmac)
			Cookie().save('token', data.authToken, {path: '/', maxAge: 60 * 60 * 24 * 365 * 10})
			browserHistory.push('/signup/params')
		} else if (data.paidState === 0 || data.paidState === 1) {
			Carrot().auth(data.id, data.cqhmac)
			Cookie().save('token', data.authToken, {path: '/', maxAge: 60 * 60 * 24 * 365 * 10})
			browserHistory.push('/signup/pay')
		}
	}
 
	render() {
		return (
			<div className='layout layout--entry'>
				<div className='grid entry-header'>
					<div className='grid__cell todayme-logo'>
						<LogoLink />
					</div>
				</div>
				<LoginForm onSubmit={this.onSubmit}/>
				<Background />
			</div>
		)
	}
}
const ProfileLogin = connect(null, {
	submitLogin
})(ProfileLoginComponent)
export default ProfileLogin
