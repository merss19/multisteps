// import { Action } from 'interfaces';
import submitData from './api'
import {LoginFormData, StateProfileLogin, Action} from './interfaces'
import * as Immutable from 'immutable'
import * as _ from 'lodash'
 
// Constant
export const LOGIN_REQUEST = 'LOGIN_REQUEST'
export const LOGIN_SUCCEESS = 'LOGIN_SUCCEESS'
export const LOGIN_FAILURE = 'LOGIN_FAILURE'

// Action Creators
const request = (): Action => ({
	type: LOGIN_REQUEST,
})
const success = (data): Action => ({
	type: LOGIN_SUCCEESS,
	data
})
 
const failure = (): Action => ({
	type: LOGIN_FAILURE,
})
interface RequestActions<T> {
	request(): void
	success(payload?: T): void
	failure(payload?: T): void
}

export const submitLogin = (data: LoginFormData, cb) => dispatch => {
	dispatch(request())
	submitData(data)
		.then(response => response.json())
		.then(res => {
			console.log(res)
			console.log('ressss')
			const data = res.data
			if (data && res.errorCode === 1) {
				dispatch(success(data))
				cb(data)
			} else {
				dispatch(failure())
			}
		})
		.catch((err) => {
			throw new Error(err.message)
		})
}

// Reducer
const initialState: StateProfileLogin = {
	isFetching: false,
	isLoad: false,
	data: {},
	isError: false
}
const initialStateImmutable = Immutable.fromJS(initialState)
 
export default (state = initialStateImmutable, action) => {
	switch (action.type) {
		case LOGIN_REQUEST:
			return state.merge({
				isFetching: true,
				isError: false,
				isLoad: false
			})
		// const updatedState2 =  state
		// return _.isEqual(updatedState2, state) ? state : updatedState2
 
		case LOGIN_SUCCEESS:
			return state.merge({
				isFetching: false,
				isLoad: true,
				isError: false,
				data: action.data
			})
		//const updatedState =  state
		//return _.isEqual(updatedState, state) ? state : updatedState
		case LOGIN_FAILURE:
			return state.merge({
				isFetching: false,
				isLoad: true,
				isError: true
			})
		default:
			return state
	}
}
